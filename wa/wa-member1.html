get_data <- function(city, variable_list){
  if(city == "San Francisco"){
    return(get_acs(
      state = "CA",
      county = "San Francisco",
      geography = "tract",
      variables = variable_list,
      geometry = TRUE,
      year = 2021
    )
    )
  }
  else{
    return(get_acs(
      state = "AZ",
      county = "Maricopa",
      geography = "tract",
      variables = variable_list,
      geometry = TRUE,
      year = 2021
    )
    )
  }
}

get_data <- function(city, variable_list){
  if(city == "San Francisco"){
    return(get_acs(
      state = "CA",
      county = "San Francisco",
      geography = "tract",
      variables = variable_list,
      geometry = TRUE,
      year = 2021
    )
    )
  }
  else{
    return(get_acs(
      state = "AZ",
      county = "Maricopa",
      geography = "tract",
      variables = variable_list,
      geometry = TRUE,
      year = 2021
    )
    )
  }
}

white_labels <- c()

  white_labels <- c(
      white_labels, paste0(
        round(white_ni[i],3)*100,
        "% to ",

        "%"
      )
  )
}
